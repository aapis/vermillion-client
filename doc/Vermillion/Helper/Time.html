def self.formatted(time = nil)
  time = ::Time.now if time.nil?
  time.strftime("%e/%-m/%Y @ %I:%M:%S%P")
end

def self.human_readable(start, finish)
  seconds = finish.to_f - start.to_f

  if seconds < 60
    "No time at all!"
  else
    minutes = (seconds / 60).round(1)
    if minutes < 1
      "#{minutes} minute"
    else
      "#{minutes} minutes"
    end
  end
end
